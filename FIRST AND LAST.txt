1. Display the lowest marks of each subject.
SELECT SUBJECT,MIN(MARKS) KEEP(DENSE_RANK FIRST ORDER BY MARKS) WORST FROM STUDENT GROUP BY SUBJECT;

2. Display the highest marks of each subject.
SELECT ROLL_NO, NAME, MARKS, MAX(MARKS) KEEP (DENSE_RANK LAST ORDER BY MARKS) OVER (PARTITION BY NAME) “BEST” FROM STUDENT ORDER BY NAME, MARKS;
